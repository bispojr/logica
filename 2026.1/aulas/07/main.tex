\section{Exercício de Fixação}

\begin{frame}{Proposição}
\begin{block}{Exercício}
\[
(\sim t \leftrightarrow x \rightarrow y)\mbox{ }\vee \sim y
\]
\end{block}

\pause

\begin{block}{Objetivo}
Construir a tabela-verdade considerando corretamente a precedência.
\end{block}
\end{frame}

%==================================================

\begin{frame}{Passo 1: Ordem de precedência}
\begin{block}{Regra}
\begin{enumerate}
\item $\sim$
\item $\wedge$, $\vee$
\item $\rightarrow$
\item $\leftrightarrow$
\end{enumerate}
\end{block}

\pause

\begin{block}{Decisão importante}
\[
\sim t \leftrightarrow x \rightarrow y
\]
\pause

é interpretado como:
\[
\sim t \leftrightarrow (x \rightarrow y)
\]
\end{block}
\end{frame}
