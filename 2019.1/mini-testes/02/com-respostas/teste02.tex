	\item (5,0 pt) {\bf [IpC Q030]}	Dada uma proposição $p$ qualquer, \\qual das declarações abaixo é {\bf falsa}? {\bf Justifique a sua resposta!}
	\begin{enumerate}
		\item A contrapositiva da contrapositiva de $p$ é $p$.
		
		\vspace*{0.1cm}
		
		{\color{blue} {\bf Verdadeira} - Admita que $p$ seja a condicional $r \rightarrow t$. \\Logo, a contrapositiva de $r \rightarrow t$ é 
			\begin{center}
				$\sim t \rightarrow \sim r$
			\end{center} 
		e a contrapositiva de $\sim t \rightarrow \sim r$ é 
		\begin{center}
			$\sim \sim r \rightarrow \sim \sim t$
		\end{center} 
		que é equivalente a $r \rightarrow t$.} 
	
		\item A contrária da contrária de $p$ é $p$.
		
			\vspace*{0.1cm}
		
		{\color{blue} {\bf Verdadeira} - Admita que $p$ seja a condicional $r \rightarrow t$. \\Logo, a contrária de $r \rightarrow t$ é 
			\begin{center}
				$\sim r \rightarrow \sim t$
			\end{center} 
			e a contrária de $\sim r \rightarrow \sim t$ é 
			\begin{center}
				$\sim \sim r \rightarrow \sim \sim t$
			\end{center} 
			que é equivalente a $r \rightarrow t$.}
		
		\item Se $V(p) = F$, então \\o valor lógico da recíproca de $p$ é verdadeiro.
		
		\vspace*{0.1cm}
		
		{\color{blue} {\bf Verdadeira} - Admita que $p$ seja a condicional $r \rightarrow t$. \\ Logo, a recíproca de $r \rightarrow t$ é 
			\begin{center}
				$t \rightarrow r$
			\end{center} 
			Assim, se $V(p) = F$, então temos que $V(r) = V$ e $V(t) = F$. \\Desta forma, $V(t \rightarrow r) = F \rightarrow V = V$.}
		
		\item Se $V(p) = F$, então \\o valor lógico da contrapositiva de $p$ é verdadeiro.
		
		\vspace*{0.1cm}
		
		{\color{red} {\bf Falsa} - Admita que $p$ seja a condicional $r \rightarrow t$. \\ Logo, a contrapositiva de $r \rightarrow t$ é 
			\begin{center}
				$\sim t \rightarrow \sim r$
			\end{center} 
			Assim, se $V(p) = F$, então temos que $V(r) = V$ e $V(t) = F$. \\Desta forma, 
			\begin{center}
				$V(\sim t \rightarrow \sim r) = \sim F \rightarrow \sim V = V \rightarrow F = F$
			\end{center}.}
		
	\end{enumerate}
