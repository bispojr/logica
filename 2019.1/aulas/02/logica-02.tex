	\begin{frame}{Conceito de Proposição}
		\begin{block}{Proposição}
			Todo o conjunto de palavras ou símbolos que exprimem um pensamento de sentido completo.
		\end{block} \pause
		\begin{block}{Características das Proposições}
				\begin{itemize}[label=-] \pause
					\item Afirmam fatos; e  \pause
					\item Exprimem juízos.
				\end{itemize}
		\end{block} \pause
		\begin{block}{Exemplos}
			\begin{itemize}[label=-]
				\item A Lua é um satélite da Terra.
				\item Recife é a capital de Pernambuco.
				\item $\pi > 5$
				\item sen $\dfrac{\pi}{2} = 1$
			\end{itemize}
		\end{block}
\end{frame}

	\begin{frame}{Questão 001}

	\begin{frame}{Princípios da Lógica}
		\begin{block}{Princípio da Não-Contradição}
			Uma proposição não pode ser verdadeira e falsa ao mesmo tempo.
		\end{block} \pause
		\begin{block}{Princípio do Terceiro Excluído}
			Toda a proposição ou é verdadeira ou é falsa, isto é, verifica-se sempre um destes casos e nunca um terceiro.
		\end{block} \pause
		\begin{exampleblock}{Por isso...}
			A Lógica Matemática é chamada de uma lógica bivalente.
		\end{exampleblock}
	\end{frame}

	\begin{frame}{Princípios da Lógica}
		\begin{block}{Exemplos}
			\begin{itemize}[label=-] 
				\item A Lua é um satélite da Terra. \pause {\color{blue} [Verdadeira]}
				\item Vasco da Gama descobriu o Brasil. \pause {\color{red} [Falsa]}
				\item Recife é a capital de Pernambuco. \pause {\color{blue} [Verdadeira]}
				\item Dante escreveu ``os Lusíadas''. \pause {\color{red} [Falsa]}
			\end{itemize}
		\end{block}

	\begin{frame}{Tipos de Proposições}
		\begin{block}{Proposições Simples (Atômica)}
			É aquela que não contém nenhuma outra proposição como parte integrante de si mesma.
		\end{block} \pause
		\begin{block}{Representação}
			Normalmente letras latinas minúsculas $p$, $q$, $r$, ...
		\end{block} \pause
		\begin{exampleblock}{Exemplos}
			\begin{itemize}[label=-]
				\item $p$: Carlos é careca.
				\item $q$: Pedro é estudante.
				\item $r$: O número 25 é quadrado perfeito.
			\end{itemize}
		\end{exampleblock}
	\end{frame}

	\begin{frame}{Tipos de Proposições}
		\begin{block}{Proposição Composta (Molecular)}
			É aquela formada pela combinação de duas ou mais proposições.
		\end{block} \pause
		\begin{block}{Representação}
			Normalmente letras latinas maiúsculas $P$, $Q$, $R$, ...
		\end{block} \pause
		\begin{exampleblock}{Exemplos}
			\begin{itemize}[label=-]
				\item $P$: Carlos é careca {\bf e} Pedro é estudante.
				\item $Q$: Carlos é careca {\bf ou} Pedro é estudante.
				\item $R$: {\bf Se} Carlos é careca, {\bf então} é infeliz.
			\end{itemize}
		\end{exampleblock} \pause
		\begin{block}{Outros nomes...}
			Fórmula proposicional ou fórmulas.
		\end{block}
	\end{frame}

	\begin{frame}{Conectivos}
		\begin{block}{Definição}
			São palavras que se usam para formar novas proposições a partir de outras.
		\end{block} \pause
		\begin{exampleblock}{Exemplos} \pause
			\begin{itemize}[label=-]
				\item O número 6 é par {\bf e} o número 8 é cubo perfeito. \pause
				\item {\bf Não} está chovendo. \pause
				\item {\bf Se} Jorge é engenheiro, {\bf então} sabe Matemática. \pause
				\item O triângulo ABC é equilátero {\bf se, e somente se,} é equiângulo.
			\end{itemize}
		\end{exampleblock}
	\end{frame}

	\begin{frame}{Valor Lógico}
		\begin{block}{Notação}
			O valor lógico de uma proposição simples $p$ indica-se por $V(p)$.
		\end{block} \pause
		\begin{block}{Assim...}
			\begin{itemize}[label=-]
				\item Se $p$ é verdadeira, então $V(p) = V$; \pause
				\item Se $p$ é falsa, então $V(p) = F$.
			\end{itemize}
		\end{block} \pause
		\begin{exampleblock}{Exemplos}
			\begin{itemize}[label=-]
				\item $p$: O Sol é verde.
				\item $q$: Um hexágono tem 9 diagonais.
				\item 2 é raiz da equação $x^2 + 3x -4 = 0$.
			\end{itemize} \pause
		
			\begin{center}
				$V(p) = F$, \ \ \ $V(q) = V$, e \ \ \ $V(r)= F$.
			\end{center}
			
		\end{exampleblock}
	\end{frame}
