	\begin{frame}{Equivalência Lógica}
		\begin{block}{Definição}
			Diz-se que uma proposição $P(p,q,r, \ldots)$ logicamente equivalente ou apenas {\bf equivalente} a uma proposição $Q(p,q,r, \ldots)$, se as tabelas-verdade destas duas proposições forem idênticas.
		\end{block} \pause
		\begin{block}{Notação}
			$P(p, q, r, \ldots) \Leftrightarrow Q(p,q,r, \ldots)$
		\end{block} \pause
		\begin{block}{Exemplos}
			\begin{itemize}
				\item $\sim \sim p \Leftrightarrow p$ \ \ e \ \ $p \rightarrow q \Leftrightarrow$ $\sim p \vee q$
			\end{itemize}
		\end{block}
	\end{frame}

	\begin{frame}{Questão 028}
		\begin{block}{[Q028]}
			Qual das alternativas abaixo é {\bf falsa}? 
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $p \wedge \sim q$ $\Leftrightarrow$ $p$ $\wedge \sim q$
			\item $p \wedge (p \vee q)$ $\Leftrightarrow$ $p$
			\item $p \rightarrow q$ $\Leftrightarrow$ $q \rightarrow p$
			\item $p \vee q$ $\Leftrightarrow$ $q \vee p$
		\end{enumerate}
	\end{frame}

	\begin{frame}{Propriedades}
		\begin{block}{Reflexiva}
			$P(p, q, r, \ldots) \Leftrightarrow P(p,q,r, \ldots)$
		\end{block} \pause
		\begin{block}{Simétrica}
			\begin{itemize}
				\item  {\bf Se} $P(p, q, r, \ldots) \Leftrightarrow Q(p,q,r, \ldots)$  {\bf então}
				\hrule \pause
				\item $Q(p, q, r, \ldots) \Leftrightarrow P(p,q,r, \ldots)$
			\end{itemize}
		\end{block} \pause
		\begin{block}{Transitiva}
			\begin{itemize}
				\item  {\bf Se} $P(p, q, r, \ldots) \Leftrightarrow Q(p,q,r, \ldots)$ \pause
				\item  {\bf e} $Q(p, q, r, \ldots) \Leftrightarrow R(p,q,r, \ldots)$ {\bf então}
				\hrule \pause
				\item $P(p, q, r, \ldots) \Leftrightarrow R(p,q,r, \ldots)$
			\end{itemize}
		\end{block}
	\end{frame}

	\begin{frame}{Tautologias e Equivalência Lógica}
		\begin{block}{Teorema}
			\center
			$P(p,q,r, \ldots)$ $\Leftrightarrow$ $Q(p,q,r,\ldots)$ \\se, e somente se a bicondicional \\$P(p,q,r, \ldots)$ $\leftrightarrow$ $Q(p,q,r,\ldots)$ \\for uma tautologia.			
		\end{block} \pause
		\begin{block}{Corolário}
			\center
			Se $P(p,q,r, \ldots)$ $\Leftrightarrow$ $Q(p,q,r,\ldots)$ então\\
			$P(P_0,Q_0,R_0, \ldots)$ $\Leftrightarrow$ $Q(P_0,Q_0,R_0,\ldots)$\\
			quaisquer que sejam $P_0,Q_0,R_0, \ldots$.
		\end{block} \pause
		\begin{exampleblock}{Nota...}
			Os símbolos $\leftrightarrow$ e $\Leftrightarrow$ são distintos. O primeiro é uma operação lógica, enquanto o segundo é uma relação lógica.
		\end{exampleblock}
	\end{frame}

	\begin{frame}{Questão 029}
	\begin{block}{[Q029]}
		Se \begin{center}
			$(p \vee q) \rightarrow r \Leftrightarrow r$
		\end{center} e 
		\begin{center}
			$r \Leftrightarrow (s \leftrightarrow t) \wedge v$,
		\end{center} {\bf não} se pode afirmar que... 
	\end{block}
	\begin{enumerate}[label=(\Alph*)]
		\item $(p \vee q) \rightarrow r \Leftrightarrow (s \leftrightarrow t) \wedge v$
		\item $(p \vee q) \Leftrightarrow (s \leftrightarrow t)$
		\item $(p \vee q) \rightarrow r \Leftrightarrow (p \vee q) \rightarrow r$
		\item $(s \leftrightarrow t) \wedge v \Leftrightarrow r$
	\end{enumerate}
\end{frame}
