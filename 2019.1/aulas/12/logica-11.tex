			\end{itemize}
		\end{exampleblock}
	\end{frame}

	\begin{frame}{Questão 030}
		\begin{block}{[Q030]}
			Dada uma proposição $p$ qualquer, \\qual das declarações abaixo é {\bf falsa}? 
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item A contrapositiva da contrapositiva de $p$ é $p$.
			\item A contrária da contrária de $p$ é $p$.
			\item Se $V(p) = F$, então \\o valor lógico da recíproca de $p$ é verdadeiro.
			\item Se $V(p) = F$, então \\o valor lógico da contrapositiva de $p$ é verdadeiro.
		\end{enumerate}
	\end{frame}

	

\begin{frame}{Negação Conjunta de Duas Proposições}
	\begin{block}{Definição}
		Chama-se de {\bf negação conjunta} de duas proposições $p$ e $q$ a proposição ``não p e não q''.
	\end{block}
	\begin{block}{Notação}
		$p \downarrow q$ $\Leftrightarrow$ $\sim p$ $\wedge \sim q$
	\end{block}
	\begin{block}{Semântica}
		\begin{center}
			\begin{tabular}{|cc|c|}
				\hline
				$p$	&	$q$	&	$p \downarrow q$	\\
				\hline
				V	&	V	&	F	\\
				V	&	F	&	F	\\
				F	&	V	&	F	\\
				F	&	F	&	V	\\
				\hline
			\end{tabular}
		\end{center}
	\end{block}
\end{frame}

\begin{frame}{Questão 031}
\begin{block}{[Q031]}
	$p \downarrow p$ é equivalente a...
\end{block}
\begin{enumerate}[label=(\Alph*)]
	\item $\sim p$
	\item $p \vee p$
	\item $p \wedge p$
	\item $p \rightarrow p$
\end{enumerate}
\end{frame}

\begin{frame}{Negação Disjunta de Duas Proposições}
\begin{block}{Definição}
	Chama-se de {\bf negação disjunta} de duas proposições $p$ e $q$ a proposição ``não p ou não q''.
\end{block}
\begin{block}{Notação}
	$p \uparrow q$ $\Leftrightarrow$ $\sim p$ $\vee \sim q$
\end{block}
\begin{block}{Semântica}
	\begin{center}
		\begin{tabular}{|cc|c|}
			\hline
			$p$	&	$q$	&	$p \uparrow q$	\\
			\hline
			V	&	V	&	F	\\
			V	&	F	&	V	\\
			F	&	V	&	V	\\
			F	&	F	&	V	\\
			\hline
		\end{tabular}
	\end{center}
\end{block}
\end{frame}

\begin{frame}{Questão 032}
\begin{block}{[Q032]}
	$\sim (p \uparrow p)$ é equivalente a...
\end{block}
\begin{enumerate}[label=(\Alph*)]
	\item $\sim p$
	\item $p$ $\vee \sim p$
	\item $p \wedge p$
	\item $p \rightarrow p$
\end{enumerate}
\end{frame}

\begin{frame}{Questão 033}
\begin{block}{[Q033]}
	Quando se afirma, em relação a $p \rightarrow q$, que \\``a recíproca também é verdadeira'', está se afirmando que...
\end{block}
\begin{enumerate}[label=(\Alph*)]
	\item $p$ e $q$ são verdadeiros.
	\item $p \Leftrightarrow q$
	\item $p \vee q$ é verdadeiro.
	\item $p \wedge q$ é falso.
\end{enumerate}
\end{frame}
