	\begin{frame}{Questão 020}	
		\begin{block}{[Gersting]}
			Qual das proposições abaixo é uma tautologia?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $P \rightarrow Q$
			\item $A \rightarrow B \rightarrow C$
			\item $P \vee P' \rightarrow Q'$
			\item {\color{blue} $(A \rightarrow B) \leftrightarrow (B' \rightarrow A')$}
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 021}	
		\begin{block}{[Gersting]}
			Qual das proposições abaixo é uma contradição?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $P \rightarrow Q$
			\item {\color{blue} $A \vee A' \rightarrow B \wedge B'$}
			\item $P \wedge P' \rightarrow Q'$
			\item $A \vee B \wedge A'$
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 022}	
		\begin{block}{[Gersting]}
			A equivalência entre as proposições $X$ e $Y$ é representada por...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $X \Rightarrow Y$
			\item $X \equiv Y$
			\item {\color{blue} $X \Leftrightarrow Y$}
			\item $X = Y$
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 023}	
		\begin{block}{[Gersting]}
			A equivalência $(A \wedge B)' \Leftrightarrow A' \vee B'$ é verdadeira, pois
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $(A \wedge B)'$ é igual a $A' \vee B'$
			\item $(A \wedge B)'$ é condição necessária de $A' \vee B'$
			\item $(A \wedge B)'$ é condição suficiente de $A' \vee B'$
			\item {\color{blue} $(A \wedge B)' \leftrightarrow A' \vee B'$ é uma tautologia}
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 024}	
		\begin{block}{[Gersting]}
			Um chip da memória de um microcomputador tem $2^4$ elementos de dois estados (LIGA-DESLIGA). Qual o número total de configurações LIGA-DESLIGA?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item {\color{blue} $2^{16}$}
			\item 16
			\item 8
			\item não é possível definir
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 025}	
		\begin{block}{[Gersting 1.3.25]}
			Em um determinado país, todos os habitantes são ou um contador de verdade que sempre fala a verdade ou mentirosos que sempre mentem. Viajando neste país, você encontra dois habitantes, Percival e Llewellyn. Percival diz ``Se eu for um contador de verdades, Llewellyn também é um contador de verdades''. Logo...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item Percival fala a verdade
			\item {\color{blue} Ambos falam a verdade}
			\item Llewellyn fala a verdade
			\item Não é possível definir
		\end{enumerate}
	\end{frame}
