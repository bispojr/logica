	\begin{frame}{Questão 026}	
		\begin{block}{[Gersting]}
			Seja $R$ a proposição $A \wedge A' \rightarrow B$. A proposição equivalente a $R$ é...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $0$
			\item {\color{blue} $1$}
			\item $A \rightarrow B$
			\item $A' \rightarrow B$
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 027}	
		\begin{block}{[Gersting]}
			Seja $R$ a proposição $P \vee P' \rightarrow Q $. A proposição equivalente a $R$ é...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $0$
			\item $1$
			\item $P \rightarrow Q$
			\item {\color{blue} $Q$}
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 028}	
		\begin{block}{[Gersting]}
			Seja $R$ a proposição $(A \rightarrow B) \rightarrow B$. A proposição equivalente a $R$ é...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item {\color{blue} $(B' \rightarrow A') \rightarrow B$}
			\item $(B \rightarrow A) \rightarrow B$
			\item $(A \rightarrow B) \vee B$
			\item $(A \rightarrow B) \wedge B$
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 029}	
		\begin{block}{[Gersting]}
			Seja $R$ a proposição $A \vee (B \wedge C)$. A proposição equivalente a $R$ é...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $A \wedge (B \vee C)$
			\item $(A \wedge B) \vee (A \wedge C)$
			\item {\color{blue} $(A \vee B) \wedge (A \vee C)$}
			\item $A \vee B \wedge C$
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 030}	
		\begin{block}{[Gersting]}
			Seja $R$ a proposição $A \rightarrow (B \rightarrow C)$. A proposição equivalente a $R$ é...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $A \rightarrow (C \rightarrow B)$
			\item $A \rightarrow (B \wedge C)$
			\item $(A \vee B) \rightarrow C$
			\item {\color{blue} $(A \wedge B) \rightarrow C$}
		\end{enumerate}
	\end{frame}
