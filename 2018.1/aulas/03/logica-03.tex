	\begin{frame}{Questão 004}	
		\begin{block}{[POSCOMP 2016]}
			Considere a seguinte proposição: 
			\begin{center}
				Todas as métricas de avaliação foram positivas.
			\end{center}
			
			A negação da proposição acima é logicamente equivalente à afirmação:
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item {\color{blue} Alguma métrica de avaliação foi negativa ou zero.}
			\item Nenhuma métrica de avaliação foi positiva.
			\item Alguma métrica de avaliação foi negativa.
			\item Todas as métricas de avaliação foram negativas ou zero.	
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 005}
		\begin{block}{[Gersting]}
			Se A é verdadeira e B é falsa, que valor você atribuiria a A $\wedge$ B?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item Verdadeira
			\item {\color{blue} Falsa}	
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 006}
		\begin{block}{[Gersting]}
			Se A é falsa e B é verdadeira, que valor você atribuiria a A $\wedge$ B?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item Verdadeira
			\item {\color{blue} Falsa}	
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 007}
		\begin{block}{[Gersting]}
			Se A e B são ambas falsas, que valor você atribuiria a A $\wedge$ B?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item  Verdadeira
			\item {\color{blue} Falsa}	
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 010}	
		\begin{block}{[Gersting]}
			``Os abacates estão maduros'' e ``Os abacates são escuros e macios'' são as proposições $A$ e $B$, respectivamente. Seja a proposição ``Os abacates só estão maduros quando estão escuros e macios''. Ela pode ser representada por...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $A \rightarrow B$
			\item {\color{blue} $B \rightarrow A$}
			\item $A \wedge B$
			\item $B \vee A$	
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 012}	
		\begin{block}{[Gersting]}
			Seja a proposição ``Se Zuckeberg estiver me espionando, os meus dados serão vendidos''. Uma vez que sabemos que Zuckeberg não está nos espionando, logicamente concluímos que...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item Os meus dados serão vendidos
			\item Os meus dados não serão vendidos
			\item A proposição é falsa
			\item {\color{blue} Não podemos saber se os dados serão vendidos.}	
		\end{enumerate}
	\end{frame}
