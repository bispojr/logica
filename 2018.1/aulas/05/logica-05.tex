	\section{Instrução pelos Colegas}
	\begin{frame}{Questão 015}	
		\begin{block}{[Gersting]}
			Qual a expressão equivalente para $A \rightarrow C \wedge B'$?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $(A \rightarrow (C \wedge B))'$
			\item $A \rightarrow (C \wedge B)'$
			\item $(A \rightarrow C) \wedge (B')$ 
			\item {\color{blue} $A \rightarrow (C \wedge (B'))$}				
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 016}	
		\begin{block}{[Gersting]}
			Qual a expressão equivalente para $D \leftrightarrow C \wedge A' \vee B$?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item {\color{blue} $D \leftrightarrow ((C \wedge (A')) \vee B)$}
			\item $(D \leftrightarrow C) \wedge ((A') \vee B)$
			\item $(D \leftrightarrow (C \wedge A)') \vee B$ 
			\item $D \leftrightarrow ((C \wedge A)' \vee B)$				
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 017}	
		\begin{block}{[Gersting]}
			Qual a alternativa que {\bf não} corresponde a um conectivo binário na lógica proposicional?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $\wedge$
			\item $\vee$
			\item {\color{blue} $\neg$} 
			\item $\leftrightarrow$				
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 018}	
		\begin{block}{[Gersting]}
			Os valores-verdade para as proposições $A$, $B$ e $C$ são V, F e V, respectivamente. Logo, o valor-verdade da proposição $A \rightarrow C \wedge B'$ é...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item {\color{blue}  Verdadeiro}
			\item Falso
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 019}
