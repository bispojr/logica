		\begin{block}{[SELECON - 2018]}
			Considere a seguinte afirmação:
			\begin{center}
				Marta é paulista ou Carlos é mineiro.
			\end{center}
		A negação lógica dessa sentença é:
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item Marta é mineira e Carlos é paulista.
			\item Marta não é mineira ou Carlos não é paulista.
			\item {\color{blue} Marta não é paulista e Carlos não é mineiro.}
			\item Marta não é paulista ou Carlos não é mineiro.
		\end{enumerate}
	\end{frame}
